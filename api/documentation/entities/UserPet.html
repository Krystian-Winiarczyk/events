import { Entity, Column, ManyToOne, JoinTable, OneToMany, JoinColumn } from 'typeorm';
import { BaseEntity } from '../../base/BaseEntity';
import { Gender } from "../../constants/Gender";
import { User } from './User';
import { File } from "./File";

@Entity({ name: 'user_pets' })
export class UserPet extends BaseEntity {
    /**
     *  Name of the dog
     * @Required true
     * @Type string
     */
    @Column()
    name: string

    /**
     *  Number of dog passport
     * @Required true
     * @Type string
     */
    @Column()
    passportNumber: string

    /**
     *  Dog breed
     * @Required false
     * @Type string
     */
    @Column({ nullable: true })
    breed: string

    /**
     *  Dog weight
     * @Required false
     * @Type number
     */
    @Column({ nullable: true })
    weight: number

    /**
     *  Color of the dog
     * @Required false
     * @Type string
     */
    @Column({ nullable: true })
    color: string

    /**
     *  Dog breeding
     * @Required false
     * @Type string
     */
    @Column({ nullable: true })
    breeding: string

    /**
     *  Dog personality
     * @Required false
     * @Type string
     */
    @Column({ nullable: true })
    personality: string

    /**
     *  Dog gender
     * @Required false
     * @Type Gender
     */
    @Column('enum', { default: Gender.MALE, enum: Gender })
    gender: Gender;

    /**
     *  Dog description
     * @Required false
     * @Type text
     */
    @Column({ nullable: true, type: 'text' })
    description: string

    /**
     *  Date of do birthday
     * @Required false
     * @Type datetime
     */
    @Column({ nullable: true, type: 'datetime' })
    birthDate: Date;

    /**
     *  Dog owner
     * @Required true
     * @Type User
     */
    @ManyToOne(() => User, (user) => user.pets)
    @JoinTable()
    user: User

    /**
     *  Dog images
     * @Required false
     * @Type File[]
     */
    @OneToMany(() => File, file => file.pet)
    @JoinColumn()
    images: File[]

    // @OneToMany(() => EventRegistration, (eventRegistration) => eventRegistration.user-user-pets)
    // @JoinTable()
    // eventRegistrations: EventRegistration[]
}
